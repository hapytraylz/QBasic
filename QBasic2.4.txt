HIPURCHASE = 100

FOR I = 1 TO 4
    READ NAME$, DEPT$, Purchase
    IF Purchase < HIPURCHASE THEN
        SAVENAME$ = NAME$
        HIPURCHASE = Purchase
    END IF
NEXT I

PRINT "HI NAME IS"; SAVENAME$
PRINT "HI PURCHASE IS"; HIPURCHASE
